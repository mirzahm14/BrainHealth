CREATE DATABASE counseling;
\c counseling

CREATE TABLE Patient(
NIK CHAR(16) NOT NULL PRIMARY KEY,
Nama VARCHAR(20) NOT NULL,
Birthdate DATE,
Sex VARCHAR(10) NOT NULL,
Password VARCHAR(64) NOT NULL);

CREATE TABLE Psikolog(
Psikolog_ID CHAR(16) NOT NULL PRIMARY KEY,
NAMA VARCHAR(64) NOT NULL,
Birthdate DATE,
Sex VARCHAR(10) NOT NULL);

CREATE TABLE Appointment(
Appointment_ID VARCHAR(16) NOT NULL PRIMARY KEY,
NIK CHAR(16) REFERENCES patient (NIK),
Psikolog_ID CHAR(16) REFERENCES psikolog (psikolog_id),
Tanggal_pemesanan DATE,
Waktu_Konsultasi TIME,
Harga VARCHAR(24) NOT NULL,
Jenis_pembayaran VARCHAR(24) NOT NULL,
Status_pemesanan VARCHAR(24) NOT NULL);

CREATE TABLE Pembayaran(
Appointment_ID VARCHAR(16) REFERENCES appointment (appointment_id),
Tagihan VARCHAR(24) NOT NULL,
Jenis_pembayaran VARCHAR(24) NOT NULL);
